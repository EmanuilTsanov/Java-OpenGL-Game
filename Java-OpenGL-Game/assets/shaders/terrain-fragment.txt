#version 330 core

in vec2 textureCoords;
in vec3 surfaceNormal;

in vec3 toLightVector;

uniform sampler2D sampler;

uniform vec3 lightColor;

out vec4 finalColor;

void main()
{
	vec4 pixelColor = texture(sampler,textureCoords*60f);
	vec3 unitNormal = normalize(surfaceNormal);
	vec3 unitToLightVector = normalize(toLightVector);
	
	float dotProduct = dot(unitNormal, unitToLightVector);
	float brightness = max(dotProduct, 0.3);
	
	vec3 diffuse = lightColor * brightness;
	
	finalColor = vec4(diffuse, 1.0) * pixelColor * vec4(1.0f,1.0f,1.0f,1.0f);
}
