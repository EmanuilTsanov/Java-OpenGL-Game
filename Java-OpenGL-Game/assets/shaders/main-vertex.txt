#version 330 core

in vec3 vertex;
in vec2 texCoords;
in vec3 normal;

uniform mat4 modelMatrix;
uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;

out vec2 textureCoords;

void main()
{
	gl_Position = projectionMatrix * viewMatrix * modelMatrix * vec4(vertex, 1.0);
	textureCoords = texCoords;
}
